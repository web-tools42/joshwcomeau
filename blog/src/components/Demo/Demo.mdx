import { Playground, PropsTable } from 'docz'

import SliderControl from '../SliderControl';


# Demo

> An interactive demonstration of how various properties affect a rendered
> child.

## Color - multiple choice

<Playground>
  <Demo
    id="colors"
    initialValues={{
      rgb: 'red',
    }}
    controls={(values, updateValue) => (
      <MultipleChoiceControl
        id="rgb"
        label="Color"
        value={values.rgb}
        updateValue={updateValue}
      >
        <Choice id="red">Red</Choice>
        <Choice id="green">Green</Choice>
        <Choice id="blue">Blue</Choice>
      </MultipleChoiceControl>
    )}
  >
    {({ rgb }) =>
      <div
        style={{
          width: 100,
          height: 100,
          borderRadius: 6,
          backgroundColor: rgb
        }}
      />
    }
  </Demo>
</Playground>


## Color - sliders

<Playground>
  <Demo
    id="colors"
    initialValues={{
      hue: 128,
      saturation: 50,
      lightness: 70,
    }}
    controls={(values, updateValue) => (
      <>
        <SliderControl
          id="hue"
          label="Hue"
          value={values.hue}
          min={0}
          max={255}
          updateValue={updateValue}
        />
        <SliderControl
          id="saturation"
          label="Saturation"
          value={values.saturation}
          min={0}
          max={100}
          updateValue={updateValue}
        />
        <SliderControl
          id="lightness"
          label="Lightness"
          value={values.lightness}
          min={0}
          max={100}
          updateValue={updateValue}
        />
      </>
    )}
  >
    {({ hue, saturation, lightness }) =>
      <div
        style={{
          width: 100,
          height: 100,
          borderRadius: 6,
          backgroundColor: 'hsl(' + hue + ', ' + saturation + '%, ' + lightness + '%)',
        }}
      />
    }
  </Demo>
</Playground>
